二分查找

迭代 (Buttom-up)

fn binsearch(s: Vec<i32>, x: i32) -> usize {
    let mut low: usize = 0;
    let mut high = s.len() - 1;
    let mut location: usize = 0;
    while low <= high && location == 0 {
        let mid = (low + high) / 2;
        if x == s[mid] {
            location = mid + 1;
        } else if x < s[mid] {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    location
}

fn main() {
    println!("{}", binsearch(vec![1,3,5,7,9], 5));
}

递归 (Top-down)

fn binsearch(s: Vec<i32>, x: i32, mut low: usize,mut high: usize) -> usize {
    let mid = (low + high) / 2;
    if x == s[mid] {
        mid + 1
    } else if x < s[mid] {
        high = mid - 1;
        binsearch(s, x, low, high)
    } else {
        low = mid + 1;
        binsearch(s, x, low, high)
    }
}

fn main() {
    println!("{}", binsearch(vec![1,3,5,7,9], 5, 1, 5));
}

迭代

use std::cmp::min;

pub fn merge_sort(vector: &mut [i32]) {
    let mut width = 1;
    let mut items = vector.to_vec();
    let len = vector.len();
    while width < len {
        let mut i = 0;
        while i < len {
            let upper = min(i + 2 * width, len);
            let mid = min(i + width, len);
            let (mut left, mut right, mut index) = (0,0,0);
            let left_vec = &vector[i..mid];
            let right_vec = &vector[mid..upper];
            let mid_items = &mut items[i..upper];
            while left < left_vec.len() && right < right_vec.len() {
                if left_vec[left] <= right_vec[right] {
                    mid_items[index] = left_vec[left];
                    left += 1;
                } else {
                    mid_items[index] = right_vec[right];
                    right += 1;
                }
                index += 1;
            }
            if left < left_vec.len() {
                mid_items[index..].copy_from_slice(&left_vec[left..]);
            }
            if right < right_vec.len() {
                mid_items[index..].copy_from_slice(&right_vec[right..]);
            }
            vector[i..upper].copy_from_slice(&items[i..upper]);
            i += 2 * width;
        }
        width *= 2;
    }
}


fn main() {
    let mut v = [6, 5, 7, 11, 3, 2, 7, 9, 1, 11, 5].to_vec();
    merge_sort(&mut v);
    for i in 0..v.len() {
        print!("{}, ", v[i]);
    }
}

递归(懒得写了, 抄了一份. 迭代会了, 递归就简单了)

pub fn merge_sort(arr: &mut [i32]) {
    let mid = arr.len() / 2;
    if mid == 0 {                 // 1
        return;
    }

    mergesort(&mut arr[..mid]);   // 2
    mergesort(&mut arr[mid..]);

    // Create an array to store intermediate result.
    let mut ret = arr.to_vec();   // 3

    // Merge the two piles.
    merge(&arr[..mid], &arr[mid..], &mut ret[..]);  // 4

    // Copy back the result back to original array.
    arr.copy_from_slice(&ret);    // 5
}

fn merge(arr1: &[i32], arr2: &[i32], ret: &mut [i32]) {
    let mut left = 0; // Head of left pile.             // 1
    let mut right = 0; // Head of right pile.
    let mut index = 0;

    // Compare element and insert back to result array.
    while left < arr1.len() && right < arr2.len() {     // 2
        if arr1[left] <= arr2[right] {                  // 3
            ret[index] = arr1[left];
            index += 1;
            left += 1;
        } else {
            ret[index] = arr2[right];
            index += 1;
            right += 1;
        }
    }

    // Copy the reset elements to returned array.
    // `memcpy` may be more performant than for-loop assignment.
    if left < arr1.len() {                              // 4
        ret[index..].copy_from_slice(&arr1[left..]);
    }
    if right < arr2.len() {
        ret[index..].copy_from_slice(&arr2[right..]);
    }
}

fn main() {
    let mut v = [6, 5, 7, 11, 3, 2, 7, 9, 1, 11, 5].to_vec();
    merge_sort(&mut v);
    for i in 0..v.len() {
        print!("{}, ", v[i]);
    }
}

快速排序(分割交换排序)

迭代

fn quick_sort(vector: &mut [i32]) {
    let mut low: usize = 0;
    let high: usize = vector.len();
    while low < high {
        let pivot_point = low;
        for index in low + 1..high {
            if vector[index] < vector[pivot_point] {
                vector.swap(index, pivot_point);
            }
        }
        low += 1;
    }
}

fn main() {
    let mut v = [15, 22, 13, 27, 12, 10, 20, 25];
    quick_sort(&mut v);
    for i in 0..v.len() {
        print!("{}, ", v[i]);
    }
}

递归

pub fn quick_sort(low: usize, high: usize, vector: &mut [i32]) {
    let mut pivot_point: usize = 0;
    if low < high {
        partition(low, high, &mut pivot_point, vector);
        quick_sort(low, pivot_point, vector);
        quick_sort(pivot_point + 1, high, vector);
    }
}

fn partition(low: usize, high: usize, pivot_point:&mut usize, vector: &mut [i32]) {
    let pivot_item = vector[low];
    let mut j = low;
    for i in low + 1..high {
        if vector[i] < pivot_item {
            j += 1;
           vector.swap(i, j);
        }
    }
    *pivot_point = j;
    vector.swap(*pivot_point, low);
}

fn main() {
    let mut v = [15, 22, 13, 27, 12, 10, 20, 25];
    quick_sort(0,8,&mut v);
    for i in 0..v.len() {
        print!("{}, ", v[i]);
    }
}
