fn binsearch(s: Vec<i32>, x: i32) -> usize {
    let mut low: usize = 0;
    let mut high = s.len() - 1;
    let mut location: usize = 0;
    while low <= high && location == 0 {
        let mid = (low + high) / 2;
        if x == s[mid] {
            location = mid + 1;
        } else if x < s[mid] {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    location
}

fn main() {
    println!("{}", binsearch(vec![1,3,5,7,9], 5));
}

fn binsearch(s: Vec<i32>, x: i32, mut low: usize,mut high: usize) -> usize {
    let mid = (low + high) / 2;
    if x == s[mid] {
        mid + 1
    } else if x < s[mid] {
        high = mid - 1;
        binsearch(s, x, low, high)
    } else {
        low = mid + 1;
        binsearch(s, x, low, high)
    }
}

fn main() {
    println!("{}", binsearch(vec![1,3,5,7,9], 5, 1, 5));
}

fn mergesort(vector: Vec<i32>) -> Vec<i32> {
    let mut items = vector.clone();
    if items.len() > 1 {
        let middle = items.len() / 2;

        let mut left = Vec::new();
        for i in 0..middle {
            left.push(items[i]);
        }

        let mut right = Vec::new();
        for i in middle..items.len() {
            right.push(items[i]);
        }

        left = mergesort(left);
        right = mergesort(right);

        let (mut i, mut ii, mut iii) = (0, 0, 0);

        while i < left.len() && ii < right.len() {
            if left[i] < right[ii] {
                items[iii] = left[i];
                i = i + 1;
            } else {
                items[iii] = right[ii];
                ii = ii + 1;
            }
            iii = iii + 1;
        }

        while i < left.len() {
            items[iii] = left[i];
            i = i + 1;
            iii = iii + 1;
        }

        while ii < right.len() {
            items[iii] = right[ii];
            ii = ii + 1;
            iii = iii + 1;
        }
    }
    return items;
}

fn main() {
    let v = mergesort(vec![6, 5, 7, 11, 3, 2, 7, 9, 1, 11, 5]);
    for i in 0..v.len()  {
        print!("{}, ", v[i]);
    }
}
